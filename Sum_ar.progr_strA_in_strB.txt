def calc_sum(start, end, step):
    res = []
    for i in range((end - start) // step + 1):
        res.append(start + step*i)
    return sum(res)

# задача со строками, проверку прошла
def str_cont(A, B):
    blist = list(B)
    k = 0
    res = 0
    for i in range(len(B) // len(A)):
        for a in A:
            if a in blist:
                blist.remove(a)
                k += 1
        if k < len(A):
            break
        else:
            res += 1
            k = 0
    return res
